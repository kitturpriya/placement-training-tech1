def check_numbers(num1,num2):
    #start writing your code here
    num_list=set()
    count=0
    for i in range(num1,num2+1):
        count1=0
        for j in range(num1,i):
            if(i%j==0 and j!=1):
                count1+=1
        if(count1!=0):
            count+=1
            num_list.add(i)
    return [num_list,count]

num1=2
num2=20
print(check_numbers(num1, num2))
